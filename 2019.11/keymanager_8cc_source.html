165 std::string KeyManager::getCN() const {
166  std::string not_found_cert_message = "Certificate is not found, can't extract device_id";
167  std::string cert;
168  if (config_.tls_cert_source == CryptoSource::kFile || config_.tls_cert_source == CryptoSource::kAndroid) {
169  if (!backend_->loadTlsCert(&cert)) {
170  throw std::runtime_error(not_found_cert_message);
171  }
172  } else { // CryptoSource::kPkcs11
173  if (!built_with_p11) {
174  throw std::runtime_error("Aktualizr was built without PKCS#11 support, can't extract device_id");
175  }
176  if (!(*p11_)->readTlsCert(&cert)) {
177  throw std::runtime_error(not_found_cert_message);
178  }
179  }
180 
181  StructGuard<BIO> bio(BIO_new_mem_buf(const_cast<char *>(cert.c_str()), static_cast<int>(cert.size())), BIO_vfree);
182  StructGuard<X509> x(PEM_read_bio_X509(bio.get(), nullptr, nullptr, nullptr), X509_free);
183  if (x == nullptr) {
184  throw std::runtime_error("Could not parse certificate");
185  }
186 
187  int len = X509_NAME_get_text_by_NID(X509_get_subject_name(x.get()), NID_commonName, nullptr, 0);
188  if (len < 0) {
189  throw std::runtime_error("Could not get CN from certificate");
190  }
191  boost::scoped_array<char> buf(new char[len + 1]);
192  X509_NAME_get_text_by_NID(X509_get_subject_name(x.get()), NID_commonName, buf.get(), len + 1);
193  std::string cn(buf.get());
194  return cn;
195 }

245 std::string KeyManager::generateUptaneKeyPair() {
246  std::string primary_public;
247 
248  if (config_.uptane_key_source == CryptoSource::kFile) {
249  std::string primary_private;
250  if (!backend_->loadPrimaryKeys(&primary_public, &primary_private)) {
251  bool result_ = Crypto::generateKeyPair(config_.uptane_key_type, &primary_public, &primary_private);
252  if (result_) {
253  backend_->storePrimaryKeys(primary_public, primary_private);
254  }
255  }
256  if (primary_public.empty() && primary_private.empty()) {
257  throw std::runtime_error("Could not get uptane keys");
258  }
259  } else if (config_.uptane_key_source == CryptoSource::kAndroid) {
260 #if defined(ANDROID)
261  primary_public = AndroidKeyStore::instance().getPublicKey();
262  if (primary_public.empty()) {
263  primary_public = AndroidKeyStore::instance().generateKeyPair();
264  }
265 #else
266  throw std::runtime_error("Aktualizr was built without Android support");
267 #endif
268  if (primary_public.empty()) {
269  throw std::runtime_error("Could not get uptane keys");
270  }
271  } else {
272  if (!built_with_p11) {
273  throw std::runtime_error("Aktualizr was built without pkcs11 support!");
274  }
275  // dummy read to check if the key is present
276  if (!(*p11_)->readUptanePublicKey(&primary_public)) {
277  (*p11_)->generateUptaneKeyPair();
278  }
279  // really read the key
280  if (primary_public.empty() && !(*p11_)->readUptanePublicKey(&primary_public)) {
281  throw std::runtime_error("Could not get uptane keys");
282  }
283  }
284  return primary_public;
285 }
