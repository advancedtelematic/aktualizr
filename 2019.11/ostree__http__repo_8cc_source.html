48  if (fp == -1) {

51  }
52  curlEasySetoptWrapper(easy_handle.get(), CURLOPT_WRITEDATA, &fp);

62  } else if (err != CURLE_OK) {
63  // other unexpected error

66  LOG_ERROR << "Failed to get object:" << curl_easy_strerror(err);
67  if (last_url != nullptr) {
