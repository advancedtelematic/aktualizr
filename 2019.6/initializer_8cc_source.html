101  CryptoSource::kFile);
102 
103  Json::Value data;
104  std::string device_id;
105  if (!storage_->loadDeviceId(&device_id)) {

107  return InitRetCode::kStorageFailure;

112  if (!response.isOk()) {

114  if (resp_code.isString() && resp_code.asString() == "device_already_registered") {
115  LOG_ERROR << "Device id" << device_id << "is occupied";
116  return InitRetCode::kOccupied;

119  return InitRetCode::kServerFailure;
