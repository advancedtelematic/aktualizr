29 bool doTestInit(StorageType storage_type, const std::string &device_register_state,
30  const std::string &ecu_register_state) {
31  LOG_INFO << "First attempt to initialize.";
32  TemporaryDirectory temp_dir;
33  conf.storage.type = storage_type;
34  conf.storage.path = temp_dir.Path();
35  conf.provision.expiry_days = device_register_state;
36  conf.provision.primary_ecu_serial = ecu_register_state;
37  const std::string good_url = conf.provision.server;
38  if (device_register_state == "noconnection") {
39  conf.provision.server = conf.provision.server.substr(conf.provision.server.size() - 2) + "11";
40  }
41 
42  bool result;
43  auto http = std::make_shared<HttpClient>();
44  auto store = INvStorage::newStorage(conf.storage);
45  {
46  KeyManager keys(store, conf.keymanagerConfig());
47  Initializer initializer(conf.provision, store, http, keys, {});
48  result = initializer.isSuccessful();
49  }
50  if (device_register_state != "noerrors" || ecu_register_state != "noerrors") {
51  EXPECT_FALSE(result);
52  LOG_INFO << "Second attempt to initialize.";
53  conf.provision.server = good_url;
54  conf.provision.expiry_days = "noerrors";
55  conf.provision.primary_ecu_serial = "noerrors";
56 
57  if (device_register_state == "noerrors" && ecu_register_state != "noerrors") {
58  // restore a "good" ecu serial in the ecu register fault injection case
59  // (the bad value has been cached in storage)
60  EcuSerials serials;
61  store->loadEcuSerials(&serials);
62  serials[0].first = Uptane::EcuSerial(conf.provision.primary_ecu_serial);
63  store->storeEcuSerials(serials);
64  }
65 
66  KeyManager keys(store, conf.keymanagerConfig());
67  Initializer initializer(conf.provision, store, http, keys, {});
68  result = initializer.isSuccessful();
69  }
70 
71  return result;
72 }
