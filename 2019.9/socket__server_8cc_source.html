38 void SocketServer::HandleOneConnection(int socket) {
39  // Outside the message loop, because one recv() may have parts of 2 messages
40  // Note that one recv() call returning 2+ messages doesn't work at the
41  // moment. This shouldn't be a problem until we have messages that aren't
42  // strictly request/response
43  DequeueBuffer buffer;
44 
45  while (true) { // Keep reading until we get an error
46  // Read an incomming message
47  AKIpUptaneMes_t *m = nullptr;
48  asn_dec_rval_t res;
49  asn_codec_ctx_s context{};
50  ssize_t received;
51  do {
52  received = recv(socket, buffer.Tail(), buffer.TailSpace(), 0);
53  LOG_TRACE << "Got " << received << " bytes "
54  << Utils::toBase64(std::string(buffer.Tail(), static_cast<size_t>(received)));
55  buffer.HaveEnqueued(static_cast<size_t>(received));
56  res = ber_decode(&context, &asn_DEF_AKIpUptaneMes, reinterpret_cast<void **>(&m), buffer.Head(), buffer.Size());
57  buffer.Consume(res.consumed);
58  } while (res.code == RC_WMORE && received > 0);
59  // Note that ber_decode allocates *m even on failure, so this must always be done
60  Asn1Message::Ptr msg = Asn1Message::FromRaw(&m);
61 
62  if (res.code != RC_OK) {
63  return; // Either an error or the client closed the socket
64  }
65 
66  // Figure out what to do with the message
67  Asn1Message::Ptr resp = Asn1Message::Empty();
68  switch (msg->present()) {
69  case AKIpUptaneMes_PR_getInfoReq: {
70  Uptane::EcuSerial serial = impl_->getSerial();
71  Uptane::HardwareIdentifier hw_id = impl_->getHwId();
72  PublicKey pk = impl_->getPublicKey();
73  resp->present(AKIpUptaneMes_PR_getInfoResp);
74  auto r = resp->getInfoResp();
75  SetString(&r->ecuSerial, serial.ToString());
76  SetString(&r->hwId, hw_id.ToString());
77  r->keyType = static_cast<AKIpUptaneKeyType_t>(pk.Type());
78  SetString(&r->key, pk.Value());
79  } break;
80  case AKIpUptaneMes_PR_manifestReq: {
81  std::string manifest = Utils::jsonToStr(impl_->getManifest());
82  resp->present(AKIpUptaneMes_PR_manifestResp);
83  auto r = resp->manifestResp();
84  r->manifest.present = manifest_PR_json;
85  SetString(&r->manifest.choice.json, manifest); // NOLINT
86  } break;
87  case AKIpUptaneMes_PR_putMetaReq: {
88  auto md = msg->putMetaReq();
89  Uptane::RawMetaPack meta_pack;
90  if (md->image.present == image_PR_json) {
91  meta_pack.image_root = ToString(md->image.choice.json.root); // NOLINT
92  meta_pack.image_targets = ToString(md->image.choice.json.targets); // NOLINT
93  meta_pack.image_snapshot = ToString(md->image.choice.json.snapshot); // NOLINT
94  meta_pack.image_timestamp = ToString(md->image.choice.json.timestamp); // NOLINT
95  } else {
96  LOG_WARNING << "Images metadata in unknown format:" << md->image.present;
97  }
98 
99  if (md->director.present == director_PR_json) {
100  meta_pack.director_root = ToString(md->director.choice.json.root); // NOLINT
101  meta_pack.director_targets = ToString(md->director.choice.json.targets); // NOLINT
102  } else {
103  LOG_WARNING << "Director metadata in unknown format:" << md->director.present;
104  }
105  bool ok;
106  try {
107  ok = impl_->putMetadata(meta_pack);
108  } catch (Uptane::SecurityException &e) {
109  LOG_WARNING << "Rejected metadata push because of security failure" << e.what();
110  ok = false;
111  }
112  resp->present(AKIpUptaneMes_PR_putMetaResp);
113  auto r = resp->putMetaResp();
114  r->result = ok ? AKInstallationResult_success : AKInstallationResult_failure;
115  } break;
116  case AKIpUptaneMes_PR_sendFirmwareReq: {
117  auto fw = msg->sendFirmwareReq();
118  auto fw_data = std::make_shared<std::string>(ToString(fw->firmware));
119  auto fut = std::async(std::launch::async, &Uptane::SecondaryInterface::sendFirmware, impl_, fw_data);
120  resp->present(AKIpUptaneMes_PR_sendFirmwareResp);
121  auto r = resp->sendFirmwareResp();
122  r->result = fut.get() ? AKInstallationResult_success : AKInstallationResult_failure;
123  } break;
124  default:
125  LOG_ERROR << "Unrecognised message type:" << msg->present();
126  return;
127  }
128 
129  // Send the response
130  if (resp->present() != AKIpUptaneMes_PR_NOTHING) {
131  int optval = 0;
132  setsockopt(socket, IPPROTO_TCP, TCP_NODELAY, &optval, sizeof(int));
133  asn_enc_rval_t encode_result =
134  der_encode(&asn_DEF_AKIpUptaneMes, &resp->msg_, Asn1SocketWriteCallback, reinterpret_cast<void *>(&socket));
135  if (encode_result.encoded == -1) {
136  return; // write error
137  }
138  optval = 1;
139  setsockopt(socket, IPPROTO_TCP, TCP_NODELAY, &optval, sizeof(int));
140  } else {
141  LOG_DEBUG << "Not sending a response to message " << msg->present();
142  }
143  } // Go back round and read another message
144 
145  // Parse error => Shutdown the socket
146  // write error => Shutdown the socket
147  // Timeout on write => shutdown
148 }
