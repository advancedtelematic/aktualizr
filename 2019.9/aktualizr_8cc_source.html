54 bool Aktualizr::UptaneCycle() {
55  result::UpdateCheck update_result = CheckUpdates().get();
56  if (update_result.updates.empty()) {
57  return true;
58  }
59 
60  result::Download download_result = Download(update_result.updates).get();
61  if (download_result.status != result::DownloadStatus::kSuccess || download_result.updates.empty()) {
62  return true;
63  }
64 
65  Install(download_result.updates).get();
66 
67  if (uptane_client_->isInstallCompletionRequired()) {
68  // If there are some pending updates then effectively either reboot (ostree) or aktualizr restart (fake pack mngr)
69  // is required to apply the update(s)
70  LOG_INFO << "About to exit aktualizr so the pending updates can be applied after reboot";
71  return false;
72  }
73 
74  if (!uptane_client_->hasPendingUpdates()) {
75  // If updates were applied and no any reboot/finalization is required then send/put manifest
76  // as soon as possible, don't wait for config_.uptane.polling_sec
77  SendManifest().get();
78  }
79 
80  return true;
81 }

83 std::future<void> Aktualizr::RunForever() {
84  std::future<void> future = std::async(std::launch::async, [&]() {
85  SendDeviceData().get();
