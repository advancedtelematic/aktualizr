145 bool DirectorRepository::matchTargetsWithImageTargets(const Uptane::Targets& image_targets) const {
146  // step 10 of https://uptane.github.io/papers/ieee-isto-6100.1.0.0.uptane-standard.html#rfc.section.5.4.4.2
147  // TODO(OTA-4800): support delegations. Consider reusing findTargetInDelegationTree(),
148  // but it would need to be moved into a common place to be resued by Primary and Secondary.
149  // Currently this is only used by aktualizr-secondary, but according to the
150  // Standard, "A Secondary ECU MAY elect to perform this check only on the
151  // metadata for the image it will install".
152  const auto& image_target_array = image_targets.targets;
153  const auto& director_target_array = targets.targets;
154 
155  for (const auto& director_target : director_target_array) {
156  auto found_it = std::find_if(
157  image_target_array.begin(), image_target_array.end(),
158  [&director_target](const Target& image_target) { return director_target.MatchTarget(image_target); });
159 
160  if (found_it == image_target_array.end()) {
161  return false;
162  }
163  }
164 
165  return true;
166 }
