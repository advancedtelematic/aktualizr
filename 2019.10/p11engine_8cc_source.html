167 bool P11Engine::readUptanePublicKey(std::string* key_out) {
168  if (config_.module.empty()) {
169  return false;
170  }
171  if ((config_.uptane_key_id.length() % 2) != 0u) {
172  return false; // id is a hex string
173  }
174 
175  PKCS11_SLOT* slot = findTokenSlot();
176  if (slot == nullptr) {
177  return false;
178  }
179 
180  PKCS11_KEY* keys;
181  unsigned int nkeys;
182  int rc = PKCS11_enumerate_public_keys(slot->token, &keys, &nkeys);
183  if (rc < 0) {
184  LOG_ERROR << "Error enumerating public keys in PKCS11 device: " << ERR_error_string(ERR_get_error(), nullptr);
185  return false;
186  }
187  PKCS11_KEY* key = nullptr;
188  {
189  std::vector<unsigned char> id_hex;
190  boost::algorithm::unhex(config_.uptane_key_id, std::back_inserter(id_hex));
191 
192  for (unsigned int i = 0; i < nkeys; i++) {
193  // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-pointer-arithmetic)
194  if ((keys[i].id_len == config_.uptane_key_id.length() / 2) &&
195  // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-pointer-arithmetic)
196  (memcmp(keys[i].id, id_hex.data(), config_.uptane_key_id.length() / 2) == 0)) {
197  // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-pointer-arithmetic)
198  key = &keys[i];
199  break;
200  }
201  }
202  }
203  if (key == nullptr) {
204  LOG_ERROR << "Requested public key was not found";
205  return false;
206  }
207  StructGuard<EVP_PKEY> evp_key(PKCS11_get_public_key(key), EVP_PKEY_free);
208  StructGuard<BIO> mem(BIO_new(BIO_s_mem()), BIO_vfree);
209  PEM_write_bio_PUBKEY(mem.get(), evp_key.get());
210 
211  char* pem_key = nullptr;
212  long length = BIO_get_mem_data(mem.get(), &pem_key); // NOLINT(google-runtime-int)
213  key_out->assign(pem_key, static_cast<size_t>(length));
214 
215  return true;
216 }

218 bool P11Engine::generateUptaneKeyPair() {
219  PKCS11_SLOT* slot = findTokenSlot();
220  if (slot == nullptr) {
221  return false;
222  }
223 
224  std::vector<unsigned char> id_hex;
225  boost::algorithm::unhex(config_.uptane_key_id, std::back_inserter(id_hex));
226 
227  // Manually generate a key and store it on the HSM
228  // Note that libp11 has a dedicated function marked as deprecated, it
229  // worked the same way in version <= 0.4.7 but tries to generate the
230  // RSA key directly on the HSM from 0.4.8. As it would not work reliably
231  // with openssl 1.1, we reimplemented it here.
232  StructGuard<EVP_PKEY> pkey = Crypto::generateRSAKeyPairEVP(KeyType::kRSA2048);
233  if (pkey == nullptr) {
234  LOG_ERROR << "Error generating keypair on the device:" << ERR_error_string(ERR_get_error(), nullptr);
235  return false;
236  }
237 
238  if (PKCS11_store_private_key(slot->token, pkey.get(), nullptr, id_hex.data(), id_hex.size()) != 0) {
239  return false;
240  }
241  if (PKCS11_store_public_key(slot->token, pkey.get(), nullptr, id_hex.data(), id_hex.size()) != 0) {
242  return false;
243  }
244 
245  return true;
246 }
