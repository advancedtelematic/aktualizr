16  * \par Description:
17  * A writeback handler for the curl library. It handles writing response
18  * data from curl into a string.
19  * https://curl.haxx.se/libcurl/c/CURLOPT_WRITEFUNCTION.html
20  *
21  */
22 static size_t writeString(void* contents, size_t size, size_t nmemb, void* userp) {
23  assert(contents);
24  assert(userp);
25  // append the writeback data to the provided string
26  auto* arg = static_cast<WriteStringArg*>(userp);
27  if (arg->limit > 0) {
28  if (arg->out.length() + size * nmemb > static_cast<uint64_t>(arg->limit)) {
29  return 0;
30  }
31  }
32  (static_cast<WriteStringArg*>(userp))->out.append(static_cast<char*>(contents), size * nmemb);
33 
34  // return size of written data
35  return size * nmemb;
36 }

191 HttpResponse HttpClient::perform(CURL* curl_handler, int retry_times, int64_t size_limit) {
192  WriteStringArg response_arg;
193  response_arg.limit = size_limit;
194  curlEasySetoptWrapper(curl_handler, CURLOPT_WRITEDATA, static_cast<void*>(&response_arg));
195  CURLcode result = curl_easy_perform(curl_handler);
196  long http_code; // NOLINT(google-runtime-int)
197  curl_easy_getinfo(curl_handler, CURLINFO_RESPONSE_CODE, &http_code);
198  HttpResponse response(response_arg.out, http_code, result, (result != CURLE_OK) ? curl_easy_strerror(result) : "");
199  if (response.curl_code != CURLE_OK || response.http_status_code >= 500) {
200  std::ostringstream error_message;
201  error_message << "curl error " << response.curl_code << " (http code " << response.http_status_code
202  << "): " << response.error_message;
203  LOG_ERROR << error_message.str();
204  if (retry_times != 0) {
205  sleep(1);
206  response = perform(curl_handler, --retry_times, size_limit);
207  }
208  }
209  LOG_TRACE << "response http code: " << response.http_status_code;
210  LOG_TRACE << "response: " << response.body;
211  return response;
212 }
