141 bool ManagedSecondary::sendFirmware(const std::shared_ptr<std::string> &data) {
142  std::lock_guard<std::mutex> l(install_mutex);
143 
144  if (expected_target_name.empty()) {
145  return false;
146  }
147  if (!detected_attack.empty()) {
148  return false;
149  }
150 
151  if (data->size() > static_cast<size_t>(expected_target_length)) {
152  detected_attack = "overflow";
153  return false;
154  }
155 
156  std::vector<Uptane::Hash>::const_iterator it;
157  for (it = expected_target_hashes.begin(); it != expected_target_hashes.end(); it++) {
158  if (it->TypeString() == "sha256") {
159  if (boost::algorithm::to_lower_copy(boost::algorithm::hex(Crypto::sha256digest(*data))) !=
160  boost::algorithm::to_lower_copy(it->HashString())) {
161  detected_attack = "wrong_hash";
162  return false;
163  }
164  } else if (it->TypeString() == "sha512") {
165  if (boost::algorithm::to_lower_copy(boost::algorithm::hex(Crypto::sha512digest(*data))) !=
166  boost::algorithm::to_lower_copy(it->HashString())) {
167  detected_attack = "wrong_hash";
168  return false;
169  }
170  }
171  }
172  detected_attack = "";
173  const bool result = storeFirmware(expected_target_name, *data);
174  return result;
