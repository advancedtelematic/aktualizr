258 static bool MatchTargetVector(const std::vector<Uptane::Target> &v1, const std::vector<Uptane::Target> &v2) {
259  if (v1.size() != v2.size()) {
260  return false;
261  }
262  for (size_t i = 0; i < v1.size(); ++i) {

265  }
